# ---- workflow content below ----
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  lint-and-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.11"
      - name: Install
        run: |
          python -m pip install --upgrade pip
          pip install ruff pytest
      - name: Lint
        run: ruff check src tests || true
      - name: Test (smoke)
        run: pytest -q tests || echo "no tests yet"
